üîß Installation Options

npm (Recommended)
Docker
Source Code

npm Installation

# Install globally
npm install -g juro-mcp-server

# Verify installation
juro-mcp-server --version

# Start server
juro-mcp-server start

Docker Installation

# Pull the image
docker pull juro/mcp-server:latest

# Run the container
docker run -p 3000:3000 -v $(pwd):/workspace juro/mcp-server:latest

Source Code Installation

# Clone repository
git clone https://github.com/arshmakker/juro.mcp.server.git

# Install dependencies
cd juro.mcp.server
npm install

# Build the project
npm run build

# Start the server
npm start

    httpPort: 8080,
    logLevel: 'info'
  },
  compliance: {
    rulePacks: ['./rules'],
    level: 'strict',
    autoScan: true
  }
}
